<script type="text/javascript">
    $(document).ready(function () {
        function trackupdate(type, prompt, promptTitle, promptHtml){
            var validate = '/lost-track-status/validate-email';
            var email = 'aroontha@gmail.com';
            $.post(
                    validate,
                    {"emailId": email, "requestOtp": 1, "rec" : 'd9663076-afe6-4ea7-bec7-9b76321d8f5f' },
                    "application/json"
                    ).done(function (data) {
                if (!(typeof data === 'object' && data !== null)) {
                    data = JSON.parse(data);
                }
                if (data.otpCreated === true) {
                    Swal.fire({
                        title: promptTitle,
                        input: 'text',
                        html: promptHtml,
                        inputAttributes: {
                            autocapitalize: 'off',
                            placeholder: 'OTP',
                            id: 'otp-id'
                        },
                        showCancelButton: true,
                        confirmButtonText: prompt,
                        showLoaderOnConfirm: true,
                        preConfirm: (otp) => {
                            $.post(
                            validate,
                            { 
                                "emailId": 'aroontha@gmail.com', 
                                "otp" : $("#otp-id").val(),
                                "rec" : 'd9663076-afe6-4ea7-bec7-9b76321d8f5f', 
                                "action" : type
                            },
                            "application/json")
                            .done(function (data) {
                                if (!(typeof data === 'object' && data !== null)) {
                                    data = JSON.parse(data);
                                }
                                if(data.disabledTracking === true){
                                    Swal.fire({
                                        title: "Tracking Disabled!"
                                    });
                                    $("#submitId").click();
                                }else if(data.enabledTracking === true){
                                    Swal.fire({
                                        title: "Tracking Enabled!"
                                    });
                                    $("#submitId").click();
                                }else{
                                    Swal.fire({
                                        title: data.message,
                                        html : 'Please check your input and try again!'
                                    });
                                }
                            });
                        },
                        allowOutsideClick: () => !Swal.isLoading()
                    });
                }
            });

        }
        
        $("#disable-tracking").click(function (e) {
            e.preventDefault();
            trackupdate(
                "disable-tracking", 
                "Disable Tracking",
                "Disable Tracking",
                "If you have found your mobile you can disable tracking from here. Check your email address for OTP. Enter OTP and click on Disable."
            );
        });
        $("#enable-tracking").click(function (e) {
            e.preventDefault();
            trackupdate(
                "enable-tracking",
                "Enable Tracking",
                "Enable Tracking",
                "If you have lost your mobile again you can enable tracking from here. Check your email address for OTP. Enter OTP and click on Enable."
            );
        });
    });
</script>
